    Objects

    let myObject = {};

    Adding and accessing object's properties

    myObject.number = 1;

    myObject["my nickname"] = "Yu";

    myObject.numberArray = new Object();

    myObject.numberArray.num1 = 2;

    console.log(myObject.number);

    console.log(myObject["my nickname"]);

    console.log(myObject.numberArray.num1);


    Enumerating properties

    for (let key in myObject) { console.log(key + ": " + myObject[key]); }


    Deleting properties

    delete myObject["my nickname"]; delete myObject.numberArray;


    Check property existence

    if ("my nickname" in myObject)


    Accessing object's methods

    myObject.func = function() { alert(`Number: ${this.number}`); }

    myObject.func();
